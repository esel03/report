# Описание структуры проекта:

main.py - файл запуска

cli_comand.py -- read_cli() -- функция считывания из командной строки параметров

pars_csv.py -- ReadCSV() -- класс для обработки csv файла и извлечение 'сырых' данных из поступающих файлов

handler.py -- choice_report_handler() -- функция мапинга ставящая отчету класс-обработчик для генерации отчета

gen_report.py -- CoffieMedianReportHandler() -- класс генерации отчета

output.py -- print_table() -- функция формирующая вывод в терминал отчета


# Наличие мапинга и отдельного класса-обработчика позволяет минимальными правками и без труда добавить генерацию любого отчета в проект


# Тесты проверяют работоспособность парсинга csv-файла, корректность сортировки и правильное вычисления медианы - т.е проверяют корректность формирования отчета
